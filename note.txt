#include <linux/module.h>

int my_init(void);	
void my_exit(void); 	

static int __init my_init(void)		// Static make func invisible outside the kernel
{
      printk("hello!!\n");	// printk write to kernel's log
      return 0;  		// return 0 if successful
}

static void __exit my_exit(void)	// __init and __exit just increase readability
{
        printk("bye\n");
}
module_init(my_init);  	// when the module is loaded into the kernel
module_exit(my_exit);   // when the module is removed from the kernel
                                                                               
MODULE_LICENSE("GPL");
MODULE_AUTHOR("Ngu Cao Nguyen");
MODULE_DESCRIPTION("simple hello world with linux kernel!!")

2. module_param()

module_param(foo, int , 0000)
* first parameter: para's name 
* second parameter: data type
* third parameter: is the permission bits

3. MODULE_PARM_DESC(param, "") => add description in modinfo .....

4. module_param_array(name, type, num, perm)

module_param_array(name, type, num, perm);
 * The first param is the parameter's (in this case the array's) name.
 * The second param is the data type of the elements of the array.
 * The third argument is a pointer to the variable that will store the number
 * of elements of the array initialized by the user at module loading time.
 * The fourth argument is the permission bits.  

5.

static char *mystring = "blah";
- At load time we can set string, if the string is not set then it will print 'blah' as a default value
- The pointer mystring will point to the provided string (stored in kernel memory). The module can then use mystring like any normal C string.
- because permission is 0000 => the parameter cannt be read or written via sysfs at runtime. It can only be set at module load time.

module_param(mystring, charp, 0000 );  
* charp: type specifier for a char pointer, allows kernel modules to accept string parameters from userspace
* 0000: File permission bits (octal) for the sysfs entry, means no permissions for anyone. The parameter will not appear in /sys/module/<module>/parameters/

NOTE: A different approach to handle overflows the buffer:
static char mystring[64] = "blah";
module_param_string(mystring, mystring, sizeof(mystring), 0000);
